3. **`src/lib/native/appleSignIn.ts`** *(new file)* â€” small helper:
   ```ts
   import { isIOS, isNative } from "./platform";
   import { SignInWithApple, SignInWithAppleResponse } from "@capacitor-community/apple-sign-in";
   import { supabase } from "@/integrations/supabase/client";

   export const canUseNativeAppleSignIn = () => isNative() && isIOS();

   export async function nativeAppleSignIn() {
     const res: SignInWithAppleResponse = await SignInWithApple.authorize({
       clientId: "com.alexandreamine.stampaway", // iOS bundle id (acts as audience)
       redirectURI: "https://account.stampaway-app.com/auth/callback", // unused on device but required field
       scopes: "email name",
       state: crypto.randomUUID(),
     });
     const idToken = res.response.identityToken;
     if (!idToken) throw new Error("Apple did not return an identity token");
     const { data, error } = await supabase.auth.signInWithIdToken({
       provider: "apple",
       token: idToken,
     });
     if (error) throw error;
     return data;
   }
   ```
4. **`src/pages/WelcomePage.tsx`** â€” replace the existing "Continue with Apple" onClick body with:

- After the new native binary is live on TestFlight/App Store and installed on the user's device, **future tweaks to the TS helper** (e.g. changing copy, toast handling, fallback logic) **can** be shipped via Capgo without another App Store submission, as long as you don't change the plugin set or entitlements.
